var keyword 
                   
'use strict'
var a = "foo";
console.log('outside function a',a);
function display(){
  var b = "bar";
  a = "beta";
  console.log('inside function a',a);
  console.log('inside function b',b)
}
display();
var a = "beta";// this is possible in var
console.log('outside function a',a);
console.log('outside function b',b); //Uncaught ReferenceError: b is not defined

// outside function a foo
// inside function a beta
// inside function b bar
// outside function a beta
Arrow function
                   
               let num = [2,3,4]
               num.map (n => n * n) // [4, 9, 16]
Splice function
                   
               let fruits = ["apple","bannana","mango","fig","orange"];
               fruits.splice(2,4); // ["mango", "fig", "orange"]
               fruits; // ["apple", "bannana"]
Slice function
                   
               let fruits = ["apple","bannana","mango","fig","orange"];
               fruits.splice(3,5); // ["fig", "orange"]
               fruits; // ["apple", "bannana", "mango", "fig", "orange"]
Class
                   
               class Polygon{
                  constructor(height,width){
                     this.height = height;
                     this.width = width;
                  }

                  calcArea(){
                     return this.height * this.width;
                  }

                  get area(){
                     return this.calcArea();
                  }
                }

               var p = new Polygon(2,3);
               console.log(p.area); // 6
