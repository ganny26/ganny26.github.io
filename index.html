XMLHttpRequest with Promise
                   
'use strict'

function get(url){
  return new Promise(function(resolve,reject){
    var xhr = new XMLHttpRequest();
    xhr.open('GET',url);
    xhr.onload = function(){
      if(xhr.status == 200){
        resolve(xhr.response);
      }
      else {
             reject(Error(xhr.statusText));
        }
    };
      xhr.onerror = function() {
        reject(Error("Network Error"));
      };
      xhr.send();
    });
}

const URL = "https://jsonplaceholder.typicode.com/users/2";
get(URL).then(function(response) {
  var response = JSON.parse(response);
    console.log(response.name); //Ervin Howell
    console.log(response.email); //Shanna@melissa.tv
}, function(error) {
  console.error("Failed!", error);
})
Call
                   
let Person = {
  fname : "James",
  lname : "Cameron",
  displayName: function(msg){
    let message = `Welcome Mr.${this.fname},${this.lname} to ${msg}`;
    console.log (message);
  }
}

              
let person = {
  fname : "Steven",
  lname : "Spielberg"
}

//Calling display name using call
Person.displayName.call(Person,'IMAX'); //Welcome Mr.James,Cameron to IMAX
Person.displayName.call(person,['SPI','Cinemas']); //Welcome Mr.Steven,Spielberg to SPI,Cinemas
Apply
                   
function applyMessage(){
  let Person = {
    fname : "Tom",
    lname : "Cruise",
    displayName:function(){
      let message = `Welcome Mr.${this.fname},${this.lname} to ${arguments[0]} ${arguments[1]}`;
      console.log (message);
    } 
  };
  Person.displayName.apply(Person,arguments);
}

//Calling display name using apply
applyMessage('IMAX','Cinemas'); //Welcome Mr.Tom,Cruise to IMAX Cinemas
