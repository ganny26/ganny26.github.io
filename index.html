Class
                   
               class Polygon{
                  constructor(height,width){
                     this.height = height;
                     this.width = width;
                  }

                  calcArea(){
                     return this.height * this.width;
                  }

                  get area(){
                     return this.calcArea();
                  }
                }

               var p = new Polygon(2,3);
               console.log(p.area); // 6
